pipeline {
  agent any
  stages {
    stage('Unit Test') { 
        steps {
            bat 'mvn clean test'
        }
    }
    stage('Build Application') {
        steps {
            bat 'mvn clean install'
        }
    }
    stage('Deploy Standalone') {
        steps {
            bat 'winpty docker run -i -p 8092:8092 -t stand-mule'
            bat 'mvn deploy -P standalone -Dmule.home=/opt/mule' 
        }
    }
  }
}
